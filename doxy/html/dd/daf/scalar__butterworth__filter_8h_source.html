366  {

368  bool is_valid_freq1 = (fcut_1_ >= 0.0) && (fcut_1_ <= f_s_/2.0);

374  // coorect frequencies

426  void initialize(Scalar x)
427  {
428  // filter state is stored in matrix S_
429  // State vector of i-th filter is stored in i-th colum
430  n_filters_ = A_.rows();
431  n_hist_ = A_.cols();
432  S_.setZero(n_hist_, n_filters_);
433 
434  // init filter state S_ to ouput x in first few iterations
435  Vector ones(A_.cols());
436  ones.setOnes();
437  ones[0] = 0.0;
438 
439  for(size_t i = 0; i < n_filters_; ++i)
440  {
441  Scalar p = (1.0 - B_(i,0))/
442  (B_.row(i).dot(ones) - B_(i,0)*A_.row(i).dot(ones));
443  S_.col(i).setConstant(p*x);
444  }
445  }

454  Scalar filter(Scalar x)
455  {
456  // update filter
457  for(size_t i = 0; i < n_filters_; ++i)
458  {
459  S_(0,i) = x - A_.row(i)*S_.col(i);
460  x = B_.row(i)*S_.col(i);
461  }
462 
463  // shift state
464  Eigen::MatrixXd tmp = S_.topRows(n_hist_-1);
465  S_.bottomRows(n_hist_-1) = tmp;
466  return x;

472  void setupLowPass(Scalar w)
473  {
474  int M = 2; // sub filter order
475  int N = order_/M; // number of filters
476 
477  bool is_valid = (N > 0);
478  ROS_ASSERT_MSG(is_valid,
479  "Invalid filter order=%d: order needs to be > 1",
480  order_);
481 
482  // store the coefficients of i-th filter in i-th row
483  // of these two matrices
484  A_.setZero(N,M+1);
485  B_.setZero(N,M+1);
486 
487  Scalar w2 = w*w;
488  Scalar r, l, a0;
489  for(int i = 0; i < N; ++i)
490  {
491  r = std::sin(M_PI*(2.0*i+1.0)/(4.0*N));
492  l = w2 + 2.0*w*r + 1.0;
493  a0 = w2/l;
494 
495  // denumerator coefficients i-th filter
496  A_(i,1) = -2.0*(1-w2)/l;
497  A_(i,2) = (w2 - 2.0*w*r + 1.0)/l;
498 
499  // numerator coefficients i-th filter
500  B_(i,0) = 1.0*a0;
501  B_(i,1) = 2.0*a0;
