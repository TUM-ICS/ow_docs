472  void setup(const Stencil& stencil)
473  {
474  // test if step size is corect
475  bool isValidStepWidth = (h_ > 0.0);
476  ROS_ASSERT_MSG(isValidStepWidth,
477  "Invalid step width: h = %f",
478  h_);
479 
480  // test if all coefficients in the stencil are unique
481  std::vector<Stencil::Scalar> s(
482  stencil.data(),
483  stencil.data() + stencil.size());
484 
485  std::vector<Stencil::Scalar>::iterator it =
486  std::unique(s.begin(),s.end());
487 
488  bool isUnique = (it == s.end());
489  ROS_ASSERT_MSG(isUnique,
490  "Invalid stencil, contains equal entries: s = [%s]",
491  eigenToString(stencil.transpose()).c_str());
492 
493  // test if the stencil lenght fits to the desired order of the deriviative
494  bool isValidConfig = (order_ < stencil.size());
495  ROS_ASSERT_MSG(isValidConfig,
496  "Invalid configuration: stencil is too short:\n"
497  "d = %d < len(s = [%s]) = %d is not fulfilled.",
498  order_,
499  eigenToString(stencil.transpose()).c_str(),
500  int(stencil.size()));
501 
502  // save
503  s_ = stencil;
504 
505  // setup sizes
506  size_ = stencil.size();
507  accuracy_ = size_ - order_;
508 
509  fac_ = Scalar(1.0)/std::pow(h_, order_);
510 
511  // Fill up the stencil matrix S
512  // Each row is the stencil vector elementswise raised to the power of
513  // the row number minus one.
514  Matrix S(size_, size_);
515  for(int i = 0; i < size_; ++i)
516  {
517  S.row(i) = s_.cast<Scalar>()*(-1.0);
518  S.row(i) = S.row(i).array().pow(i);
519  }
520 
521  // invert S
522  Matrix S_inv = S.inverse();
523 
524  // compute the factorial of d
525  Scalar d_fact = ow::factorial(order_);
526 
527  // Vector of Kronecker deltas: all entries are zero besides
528  // the row where row-1 == d
529  // and then multiply by dFact.
530  Vector b = Vector(size_);
531  b.setZero();
532  b[order_] = d_fact;
533 
534  // now we can compute the coefficient vector via matrix vector multiplication
535  c_ = S_inv*b;
536 
537  // setup memory buffer
538  x_.resize(size_);
539  }
