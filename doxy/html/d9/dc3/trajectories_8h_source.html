242  {

251  // zero acc at begining

255  // a1 = a0 + c1/3*h_0^2

264  // Therefore the left and right modified segments overlapp

273  // Therefore the left and right modified segments don't overlapp

284  As(i,i+1) = h[i];

306  // now a1 can be computed: c1/3*h0**2 + a0 == a1
307  a[1] = a[0] + h[0]*h[0]*(c[1]/3);
308  // now a can be computed: c3*h4**2/3 + a[n] == a[n-1]
309  a[n-2] = a[n-1] + h[m-1]*h[m-1]*(c[n-2]/3);

318  coeff[3] = (1/(3*h[i]))*(c[i+1]-c[i]);
