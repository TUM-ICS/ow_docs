277 template <typename Derived1, typename Derived2>
278 inline AngularPosition slerp(
279  const ow_core::QuaternionBase<Derived1>& Qi,
280  const ow_core::QuaternionBase<Derived2>& Qf,
281  ow::Scalar s)
282 {
283  typedef ow::Scalar Scalar;
284  const Scalar one = Scalar(1) - std::numeric_limits<Scalar>::epsilon();
285 
286  ow::AngularPosition Qs;
287  Scalar d = Qi.derived().dot(Qf.derived());
288  Scalar d_abs = std::abs(d);
289 
290  Scalar scale_0;
291  Scalar scale_1;
292 
293  if(d_abs >= one)
294  {
295  // linear interpolation
296  scale_0 = Scalar(1) - s;
297  scale_1 = s;
298  }
299  else
300  {
301  // spherical interpolation
302  Scalar th = std::acos(d_abs);
303  Scalar th_sin = std::sin(th);
304 
305  scale_0 = std::sin((Scalar(1) - s)*th)/th_sin;
306  scale_1 = std::sin(s*th)/th_sin;
307  }
308  if(d < Scalar(0))
309  {
310  // take the shortest path
311  scale_1 = -scale_1;
312  }
313  Qs.coeffs() = scale_0*Qi.derived().coeffs()
314  + scale_1*Qf.derived().coeffs();
315  return Qs;
316 }
